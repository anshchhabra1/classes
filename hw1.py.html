import numpy
import random
import time

class SearchAlgorithms:
    def generate_array(n):
        return numpy.random.randint(99, size=(1, n))

    def linear_search(arrayR, x):
        start_time = time.time()
        for i in range(len(arrayR)):
            if arrayR[i] == x:
                end_time = time.time()
                return i, (end_time - start_time)
        return -1, None

    def linear_searchI(arrayR, x):
        start_time = time.time()
        pos = 0
        found = False
        while pos < len(arrayR) and not found:
            if arrayR[pos] == x:
                found = True
                end_time = time.time()
                return pos, (end_time - start_time)
            else:
                pos = pos+1
        return -1, None

    def binary_search(arrayR, x):
        start_time = time.time()
        arrayR = sorted(arrayR)
        first = 0
        last = len(arrayR) - 1
        while first <= last:
            center = (last + first) // 2
            if arrayR[center] == x:
                end_time = time.time()
                return center, end_time - start_time
            else:
                if x < arrayR[center]:
                    last = center - 1
                else:
                    first = center + 1
        return -1, None

if __name__ == "__main__":
    n = int(input("size?"))
    arrayR = 1(SearchAlgorithms.generate_array(n))
    print("array:", arrayR)
    x = random.choice(arrayR[0])
    print("randomly selected number:", x)
    linear, linearexecutiontime = SearchAlgorithms.linear_search(arrayR[0], x)
    print("iterative linear search :", linear)
    print("execution time:", linearexecutiontime, "seconds")
    sequential, sequentialexecutiontime = SearchAlgorithms.linear_searchI((arrayR[0]), x)
    print("non-iterative linear search :", sequential)
    print("execution time:", sequentialexecutiontime, "seconds")
    binary, binaryexecutiontime = SearchAlgorithms.binary_search((arrayR[0]), x)
    print("binary search :", binary)
    print("execution time:", binaryexecutiontime, "seconds")
